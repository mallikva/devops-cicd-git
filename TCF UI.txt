$ git init //Initialise Repository
$ git add . // Add file to staging so that git can track
$ git commit -m "Rationale" // Commit to local repo
$ git push -u origin main // Commit to GitHub
$ git log //log the changes
$ git revert -m 1 id from log //revert to previous versions
$ git diff previous id1 previous id 2 //difference between 2 versions
$ git status //current status of git

$ git remote add origin https://github.com/mallikva/devops-cicd-git2.git //adding the origin I mean git repo

$ git pull //to fetch the updates from GitHub repo (origin) to your local repo

$ q //to quit and start again

$ cd "directory name"//change the directory
$ cd .. // move one folder up
$ ls //list the items
$ ls -l //long listing format
$ ls -a //list hidden file
$ ls -ltr // long listing sort by modification time reverse order



ls - List directory contents
cd - Change the current directory
pwd - Print the current working directory
echo - Display a line of text
cat - Concatenate and display files
cp - Copy files and directories
mv - Move or rename files
rm - Delete files or folders
touch - Create an empty file or update its time
mkdir - Create a new folder
